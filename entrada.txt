program {
    // 1. Datos del estudiante
    float nota1 = 8.5;
    float nota2 = 9.0;
    float nota3 = 6.0;
    
    // 2. Cálculo del promedio simple
    float promedio = (nota1 + nota2 + nota3) / 3;

    // 3. Variables de estado
    bool aprobado = 0;
    int puntosExtra = 0;

    // 4. Lógica de aprobación (Si el promedio es mayor o igual a 7)
    if (promedio >= 7.0) {
        aprobado = 1;
        
        // Si además es una nota excelente, damos puntos extra
        if (promedio > 8.5) {
            puntosExtra = 5;
        }
    } else {
        aprobado = 0;
        puntosExtra = 0;
    }

    // 5. Ajuste final de nota si tuvo puntos extra
    float notaFinal = promedio + puntosExtra;
}
